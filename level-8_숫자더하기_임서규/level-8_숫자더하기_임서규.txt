public static void main(String[] args) {
	Scanner sc = new Scanner(System.in);
	System.out.println("1부터 시작되는 정수를 입력하세요");
	int n = sc.nextInt();
	input(n);
	sc.close();
}
		
public static void input(int n) {
	int num = n;
	int sum = 0;
	for(int i=0; i<=num; i++) {
		sum += i;
	}	
	System.out.print(n+"=>"+sum);
}
